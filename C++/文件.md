//将文件中的大写字母转化为小写后输出
#include<iostream>
#include<fstream>
#include<string>
using namespace std;

void ReadFile(string s); 
void Change(string s1, string s2);
int main()
{
	system("chcp 65001"); 
	ReadFile("D:\\ff.txt"); //指出绝对路径
	Change("D:\\ff.txt", "D:\\ff2.txt");
	ReadFile("D:\\ff2.txt");
	return 0;

}
void ReadFile(string s)
{
	ifstream file(s); //传过来的字符串参数本来就带有双引号，这里的s不写双引号了
	if (!file)
	{
		cout << "failed to open the file" << endl;
	}
	string s1;
	while (getline(file, s1))
	{
		cout << s1;
	}

	
}
void Change(string s1, string s2)
{
	ifstream file2(s1);
	if (!file2)
	{
		cout << "failed to open the file" << endl;
	}
	string s;
	ofstream file3;
	file3.open(s2, ios::out);
	while (getline(file2, s))
	{
		for (char& c: s)
		{
			if (c >= 'a' && c <= 'z')
			{
				c = c - 32;
			}
		}
		file3 << s;
	}

}

class Student
{
public:
	string num;
	string name;
	string sex;
	int score;
public:
	Student(string nu = "", string na = "", string se = "", int s = 0);
	friend ostream& operator<<(ostream& out, const Student& s);

};
Student::Student(string nu, string na, string se, int s)
{
	num = nu;
	name = na;
	sex = se;
	score = s;
}
ostream& operator<<(ostream& out, const Student& s)
{
	cout << s.num << " " << s.name << " " << s.sex << " " << s.score << endl;
	return out;
}

		for (Student s : stu)
